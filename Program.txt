using System;
using System.Collections.Generic;
namespace AdvancedBankingSystem
{
    // ================= MAIN PROGRAM =================
    class Program
    {
        static List<BankAccount> accounts = new List<BankAccount>();
        static void Main(string [] args)
        {
            Console.WriteLine("=== Advanced Banking System ===");

        // Sample accounts
        accounts.Add(new BankAccount("Alice", 5000, "Savings"));
        accounts.Add(new BankAccount("Bob", 3000, "Current"));
        accounts.Add(new BankAccount("Charlie", 7000, "Savings"));

        while (true)
        {
            Console.WriteLine("\n1.Create Account\n2.Deposit\n3.Withdraw\n4.Transfer\n5.Calculate Interest\n6.View Accounts\n7.Payment Schedule\n8.Recursive Compound Interest\n9.Sum Deposits\n10.Test ref/value\n11.Bank Stats\n12.Test Optional Params\n13.Exit");

            if (!int.TryParse(Console.ReadLine(), out int choice)) continue;

            switch (choice)
            {
                case 1:
                    Console.Write("Name: ");
                    string name = Console.ReadLine();
                    Console.Write("Deposit: ");
                    decimal dep = decimal.Parse(Console.ReadLine());
                    accounts.Add(new BankAccount(name, dep, "Savings"));
                    break;

                case 2:
                    Console.Write("Account #: ");
                    int a = int.Parse(Console.ReadLine());
                    Console.Write("Amount: ");
                    decimal amt = decimal.Parse(Console.ReadLine());
                    accounts[a - 1].Deposit(amt);
                    break;

                case 3:
                    Console.Write("Account #: ");
                    a = int.Parse(Console.ReadLine());
                    Console.Write("Amount: ");
                    amt = decimal.Parse(Console.ReadLine());
                    accounts[a - 1].Withdraw(amt);
                    break;

                case 4:
                    Console.WriteLine("1.Basic 2.Description 3.Notification");
                    int t = int.Parse(Console.ReadLine());

                    Console.Write("From: "); int f = int.Parse(Console.ReadLine());
                    Console.Write("To: "); int to = int.Parse(Console.ReadLine());
                    Console.Write("Amount: "); amt = decimal.Parse(Console.ReadLine());

                    if (t == 1)
                        accounts[f - 1].Transfer(accounts[to - 1], amt);
                    else if (t == 2)
                    {
                        Console.Write("Description: ");
                        string d = Console.ReadLine();
                        accounts[f - 1].Transfer(accounts[to - 1], amt, d);
                    }
                    else
                    {
                        Console.Write("Description: ");
                        string d = Console.ReadLine();
                        accounts[f - 1].Transfer(accounts[to - 1], amt, d, true);
                    }
                    break;

                case 5:
                    Console.Write("Account #: ");
                    a = int.Parse(Console.ReadLine());
                    Console.Write("Months: "); int m = int.Parse(Console.ReadLine());
                    Console.WriteLine("1.Simple 2.Custom 3.Compound");
                    int opt = int.Parse(Console.ReadLine());

                    if (opt == 1)
                        Console.WriteLine(accounts[a - 1].CalculateInterest(m));
                    else if (opt == 2)
                    {
                        Console.Write("Rate: "); decimal r = decimal.Parse(Console.ReadLine());
                        Console.WriteLine(accounts[a - 1].CalculateInterest(m, r));
                    }
                    else
                    {
                        Console.Write("Rate: "); decimal r = decimal.Parse(Console.ReadLine());
                        Console.WriteLine(accounts[a - 1].CalculateInterest(m, r, true));
                    }
                    }
                    break;

                case 6:
                    foreach (var acc in accounts) acc.DisplayAccountInfo();
                    break;

                case 7:
                    Console.Write("Months: "); int months = int.Parse(Console.ReadLine());
                    Console.Write("Payment: "); decimal pay = decimal.Parse(Console.ReadLine());
                    LoanCalculator.DisplayPaymentSchedule(months, pay);
                    break;

                case 8:
                    Console.Write("Principal: "); decimal p = decimal.Parse(Console.ReadLine());
                    Console.Write("Rate: "); decimal rate = decimal.Parse(Console.ReadLine());
                    Console.Write("Years: "); int y = int.Parse(Console.ReadLine());
                    Console.WriteLine(LoanCalculator.CompoundInterest(p, rate, y));
                    break;

                case 9:
                    decimal[] arr = { 100, 200, 300 };
                    Console.WriteLine(LoanCalculator.SumDeposits(arr, 0));
                    break;

                case 10:
                    decimal bal = 1000;
                    Console.WriteLine($"Original: {bal}");
                    TransactionProcessor.TryUpdateBalance(bal, 500);
                    Console.WriteLine($"After value: {bal}");
                    TransactionProcessor.UpdateBalance(ref bal, 500);
                    Console.WriteLine($"After ref: {bal}");
                    TransactionProcessor.ProcessTransaction(200, "Deposit", out string code, out DateTime time);
                    Console.WriteLine($"Code: {code}, Time: {time}");
                    break;

                case 11:
                    BankAccount.DisplayBankStatistics();
                    break;

                case 12:
                    // Demonstrate optional & named parameters
                    BankAccount.CreateSavingsAccount("John", 5000);
                    BankAccount.CreateSavingsAccount("Mary", 4000, 0.03m);
                    BankAccount.CreateSavingsAccount("David", 6000, 0.04m, 2000, "Lekki");
                    BankAccount.CreateSavingsAccount(accountHolder: "Chris", branch: "Ikeja", initialDeposit: 7000);
                    break;

                case 13:
                    return;
            }
        }
    }
}
